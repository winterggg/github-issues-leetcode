# 问题

https://leetcode.cn/problems/generate-parentheses/

# 笔记

简单 DFS

# 代码

## Python

```python
class Solution:
    def generateParenthesis(self, n: int) -> List[str]:
        ans = []
        def dfs(left, n, path=None):
            if not path: path = []

            if n * 2 == len(path):
                if left == n:
                    ans.append(''.join(path))
                return
            
            path.append('(')
            dfs(left + 1, n, path)
            path.pop()

            if left <= len(path) - left: return
            path.append(')')
            dfs(left, n, path)
            path.pop()

        dfs(0, n)
        return ans
```
剪枝优化一下（差别不大）：
```python
class Solution:
    def generateParenthesis(self, n: int) -> List[str]:
        ans = []
        def dfs(left, n, path=None):
            if not path: path = []

            if n * 2 == len(path):
                if left == n:
                    ans.append(''.join(path))
                return
            
            right = len(path) - left
            if left < n:
                path.append('(')
                dfs(left + 1, n, path)
                path.pop()
            
            if right < left:
                if left <= len(path) - left: return
                path.append(')')
                dfs(left, n, path)
                path.pop()

        dfs(0, n)
        return ans
```
