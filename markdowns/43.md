# 问题

https://leetcode.cn/problems/implement-queue-using-stacks/

# 笔记

两个栈，一个输入，一个输出，transfer 维护队列顺序。

# 代码

## Python

```python
class MyQueue:

    def __init__(self):
        self.__stack1, self.__stack2 = [], []   
    
    def __transfer(self):
        if not self.__stack2:
            while self.__stack1:
                self.__stack2.append(self.__stack1.pop())

    def push(self, x: int) -> None:
        self.__stack1.append(x)

    def pop(self) -> int:
        self.__transfer()
        return self.__stack2.pop()

    def peek(self) -> int:
        self.__transfer()
        return self.__stack2[-1]

    def empty(self) -> bool:
        return not(self.__stack1 or self.__stack2)



# Your MyQueue object will be instantiated and called as such:
# obj = MyQueue()
# obj.push(x)
# param_2 = obj.pop()
# param_3 = obj.peek()
# param_4 = obj.empty()
```
