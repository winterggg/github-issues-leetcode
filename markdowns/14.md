# 问题

https://leetcode.cn/problems/3sum/

# 笔记

1. 排序加双指针；
2. 注意三处滤重。

# 代码

## Python

```python
class Solution:
    def threeSum(self, nums_raw: List[int]) -> List[List[int]]:
        nums = sorted(nums_raw)
        ret = []

        for i in range(len(nums)):
            if i and nums[i] == nums[i-1]: continue

            l, r = i + 1, len(nums) - 1
            while l < r:
                sum_ = nums[i] + nums[l] + nums[r]
                if sum_ == 0:
                    ret.append([nums[i], nums[l], nums[r]])
                    l += 1
                    r -= 1

                    while l < r and nums[l] == nums[l-1]:
                        l += 1
                    while l < r and nums[r] == nums[r+1]:
                        r -= 1
                elif sum_ > 0:
                    r -= 1
                else:
                    l += 1

        return ret
```

## Java

```java
import java.util.*;

public class Solution {
    public List<List<Integer>> threeSum(int[] nums_raw) {
        // 一般来说可以省略
        int[] nums = Arrays.copyOf(nums_raw, nums_raw.length);
        Arrays.sort(nums);
        List<List<Integer>> ret = new ArrayList<>();

        for (int i = 0; i < nums.length; i++) {
            if (i > 0 && nums[i] == nums[i - 1]) continue;

            int l = i + 1;
            int r = nums.length - 1;

            while (l < r) {
                int sum = nums[i] + nums[l] + nums[r];

                if (sum == 0) {
                    ret.add(Arrays.asList(nums[i], nums[l], nums[r]));
                    l++; r--;

                    while (l < r && nums[l] == nums[l - 1]) l++;
                    while (l < r && nums[r] == nums[r + 1]) l--;
                } else if (sum > 0) r--;
                 else l++;
            }
        }

        return ret;
    }
}
```
