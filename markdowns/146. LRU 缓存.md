# 问题

https://leetcode.cn/problems/lru-cache/

# 笔记

1. 简单模拟题，但代码较长，注意做好方法抽象。
2. 用双端队列维护时间序列，用mappings实现快速查询。

# 代码

## Python

```python
NIL = -1

class Node:
    def __init__(self, value=NIL, key=NIL, left=None, right=None):
        self.l = left
        self.r = right
        self.v = value
        self.k = key
    
    @staticmethod
    def create_list():
        left = Node()
        right = Node()
        left.r = right
        right.l = left
        return left, right
    
    def delete(self):
        ret = self.k
        self.l.r = self.r
        self.r.l = self.l
        return ret
    
    def insert(self, node):
        node.l = self
        node.r = self.r
        self.r.l = node
        self.r = node

class LRUCache:

    def __init__(self, capacity: int):
        self.cap = capacity
        self.l, self.r = Node.create_list()
        self.node_map = {}

    def __lift(self, node):
        node.delete()
        self.l.insert(node)

    def get(self, key: int) -> int:
        node = self.node_map.get(key)
        if not node: return -1

        ret = node.v
        self.__lift(node)
        return ret


    def put(self, key: int, value: int) -> None:
        node = self.node_map.get(key)
        if node:
            node.v = value
            self.__lift(node)
            return
        
        node = Node(value, key)
        if len(self.node_map) == self.cap:
            del_key = self.r.l.delete()
            del self.node_map[del_key]
        self.l.insert(node)
        self.node_map[key] = node

```
