```python
class Solution:
    def lengthOfLIS(self, nums: List[int]) -> int:
        dp = [ 0 ] * len(nums)
        dp[0] = 1 
        ans = 1
        for i in range(1, len(nums)):
            max_ = 1
            for j in range(0, i):
                if nums[j] >= nums[i]: continue
                max_ = max(max_, dp[j] + 1)
            dp[i] = max_
            ans = max(ans, dp[i])
        
        return ans
```
