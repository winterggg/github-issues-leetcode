NIL = -1

class Node:
    def __init__(self, value=NIL, key=NIL, left=None, right=None):
        self.l = left
        self.r = right
        self.v = value
        self.k = key
    
    @staticmethod
    def create_list():
        left = Node()
        right = Node()
        left.r = right
        right.l = left
        return left, right
    
    def delete(self):
        ret = self.k
        self.l.r = self.r
        self.r.l = self.l
        return ret
    
    def insert(self, node):
        node.l = self
        node.r = self.r
        self.r.l = node
        self.r = node

class LRUCache:

    def __init__(self, capacity: int):
        self.cap = capacity
        self.l, self.r = Node.create_list()
        self.node_map = {}

    def __lift(self, node):
        node.delete()
        self.l.insert(node)

    def get(self, key: int) -> int:
        node = self.node_map.get(key)
        if not node: return -1

        ret = node.v
        self.__lift(node)
        return ret


    def put(self, key: int, value: int) -> None:
        node = self.node_map.get(key)
        if node:
            node.v = value
            self.__lift(node)
            return
        
        node = Node(value, key)
        if len(self.node_map) == self.cap:
            del_key = self.r.l.delete()
            del self.node_map[del_key]
        self.l.insert(node)
        self.node_map[key] = node

class Node {
    private Node left;
    private Node right;
    private int value;
    private int key;
    private static final int NIL = -1;

    public Node(int value, int key, Node left, Node right) {
        this.left = left;
        this.right = right;
        this.value = value;
        this.key = key;
    }

    public Node() {
        this(NIL, NIL, null, null);
    }

    public static Node[] createList() {
        Node left = new Node();
        Node right = new Node();
        left.setRight(right);
        right.setLeft(left);
        return new Node[]{left, right};
    }

    public int delete() {
        int ret = key;
        left.setRight(right);
        right.setLeft(left);
        return ret;
    }

    public void insert(Node node) {
        node.setLeft(this);
        node.setRight(this.right);
        this.right.setLeft(node);
        this.right = node;
    }

    public void setLeft(Node left) {
        this.left = left;
    }

    public void setRight(Node right) {
        this.right = right;
    }

    public Node getLeft() {
        return left;
    }

    public Node getRight() {
        return right;
    }

    public int getValue() {
        return value;
    }

    public void setValue(int value) {
        this.value = value;
    }

    public int getKey() {
        return key;
    }

    public void setKey(int key) {
        this.key = key;
    }

}

class LRUCache {
    private int capacity;
    private Node left;
    private Node right;
    private Map<Integer, Node> nodeMap;

    public LRUCache(int capacity) {
        this.capacity = capacity;
        Node[] nodes = Node.createList();
        this.left = nodes[0];
        this.right = nodes[1];
        this.nodeMap = new HashMap<>();
    }

    private void lift(Node node) {
        node.delete();
        left.insert(node);
    }
    
    public int get(int key) {
        Node node = nodeMap.get(key);
        if (node == null) return -1;

        int ret = node.getValue();
        lift(node);
        return ret;
    }
    
    public void put(int key, int value) {
        Node node = nodeMap.get(key);
        if (node != null) {
            node.setValue(value);
            lift(node);
            return;
        }

        node = new Node(value, key, null, null);
        if (nodeMap.size() == capacity) {
            int delKey = right.getLeft().delete();
            nodeMap.remove(delKey);
        }
        left.insert(node);
        nodeMap.put(key, node);
    }
}
