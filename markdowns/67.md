## Python

### 方法一：

```python
class Solution:
    def addMinimum(self, word: str) -> int:
        i, ans, n = 0, 0, len(word)
        while i < len(word):
            # i, i+1, i+2
            if i+2 >= n:
                k = n - i
                if k == 1: ans += 2
                else:
                    if ord(word[i]) < ord(word[i+1]):
                        ans += 1
                    else:
                        ans += 4
                break
            
            x, y, z = word[i:i+3]

            if x == 'a':
                if y == 'a': i, ans = i + 1, ans + 2
                elif y == 'b':
                    if z == 'c': i, ans = i + 3, ans
                    else: i, ans = i + 2, ans + 1
                else: i, ans = i + 2, ans + 1
            elif x == 'b':
                if y == 'c': i, ans = i + 2, ans + 1
                else: i, ans = i + 1, ans + 2
            else:
                i, ans = i + 1, ans + 2
        
        return ans
```

### 方法二：

```python
class Solution:
    def addMinimum(self, s: str) -> int:
        # s, ans = "c" + s + "a", 0 # 可以加哨兵
        ans = ord(s[0]) - ord(s[-1]) + 2
        for x, y in pairwise(map(ord, s)):
            ans += (y - x + 2) % 3
        return ans
```

### 方法三：

```python
class Solution:
    def addMinimum(self, s: str) -> int:
        t = 1 + sum(x >= y for x, y in pairwise(s))
        return t * 3 - len(s)
```

### 方法五：

```python
class Solution:
    def addMinimum(self, word: str) -> int:
        s = 'abc'
        i, j = 0, 0
        ans = 0
        while j < len(word):
            if s[i] == word[j]:
                i, j = (i + 1) % 3, j + 1
            else:
                ans += 1
                i = (i + 1) % 3
        
        ans += (3 - i) % 3
        return ans
```
