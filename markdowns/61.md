# 问题

https://leetcode.cn/problems/minimum-window-substring/

# 笔记

核心思路是使用滑动窗口算法寻找最短子串

# 代码

## Python

```python
class Solution:
    def minWindow(self, s: str, t: str) -> str:
        if len(s) < len(t): return ""

        req, wnd = defaultdict(int), defaultdict(int)
        
        # 统计t中各字符的出现次数
        for c in t:
            req[c] += 1

        # 初始化距离和左指针
        dist, left_ptr = len(req), 0
        ans = None

        # 遍历s，更新窗口和距离
        for right_ptr, char in enumerate(s):
            wnd[char] += 1
            
            # 更新距离
            if wnd[char] == req[char]:
                dist -= 1

            # 缩小窗口并更新结果
            while dist == 0:
                if not ans or len(ans) > right_ptr - left_ptr + 1:
                    ans = s[left_ptr:right_ptr + 1]

                if wnd[s[left_ptr]] == req[s[left_ptr]]:
                    dist += 1
                wnd[s[left_ptr]] -= 1
                left_ptr += 1

        return ans if ans else ""
```

wnd 其实可以省略，例如下面的样子(by chatgpt)

```python
from collections import defaultdict, Counter

class Solution:
    def minWindow(self, s: str, t: str) -> str:
        if len(t) > len(s):
            return ""

        req = Counter(t)
        dist, matched_count = len(req), 0
        ans = None
        left_ptr = 0

        for right_ptr, char in enumerate(s):
            # 如果当前字符在 req 中
            if char in req:
                req[char] -= 1
                # 当当前字符的计数仍然大于等于0时，说明窗口中的字符数增加了
                if req[char] >= 0:
                    matched_count += 1

            # 缩小窗口并更新结果
            while matched_count == len(t):
                if not ans or len(ans) > right_ptr - left_ptr + 1:
                    ans = s[left_ptr:right_ptr + 1]

                left_char = s[left_ptr]
                if left_char in req:
                    req[left_char] += 1
                    # 当左侧字符的计数大于0时，说明窗口中的字符数减少了
                    if req[left_char] > 0:
                        matched_count -= 1
                left_ptr += 1

        return ans if ans else ""

```
