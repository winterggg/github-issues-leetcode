# 问题

https://leetcode.cn/problems/sort-an-array/

# 笔记

快排有很多写法，主要的区别是partition方式，感觉第1种和第3种比较好理解。

### 1：超时，wtf？

```python
import random

class Solution:
    def sortArray(self, nums: List[int]) -> List[int]:
        self.__sort(nums, 0, len(nums) - 1)
        return nums
    
    def __sort(self, nums, l, r):
        if l >= r: return
        
        p = random.randint(l, r)
        v = nums[p]
        nums[l], nums[p] = nums[p], nums[l]

        # [l+1, i] < v, [i+1, j) >= v # 等号放在哪边都不影响正确性
        i, j = l, l + 1
        while j < len(nums):
            if nums[j] >= v:
                j += 1
            else:
                i += 1
                nums[i], nums[j] = nums[j], nums[i]
                j += 1
        
        nums[l], nums[i] = nums[i], nums[l]
        self.__sort(nums, l, i-1)
        self.__sort(nums, i+1, r)
```

```python
import random

class Solution:
    def sortArray(self, nums: List[int]) -> List[int]:
        self.__sort(nums, 0, len(nums) - 1)
        return nums
    
    def __sort(self, nums, l, r):
        if l >= r: return
        
        p = random.randint(l, r)
        v = nums[p]
        nums[l], nums[p] = nums[p], nums[l]

        # [l+1, i) <= v, (j, r] >= v
        i, j = l+1, r
        while True:
            while i <= j and nums[i] < v: i += 1 # 跳出时，nums[j] < v, 所以后面要和 j 交换！
            while i <= j and nums[j] > v: j -= 1
            if i >= j: break
            nums[i], nums[j] = nums[j], nums[i]
            i += 1
            j -= 1
            
        nums[l], nums[j] = nums[j], nums[l]
        self.__sort(nums, l, j-1)
        self.__sort(nums, j+1, r)
```

```python
import random

class Solution:
    def sortArray(self, nums: List[int]) -> List[int]:
        self.__sort(nums, 0, len(nums) - 1)
        return nums
    
    def __sort(self, nums, l, r):
        if l >= r: return
        
        p = random.randint(l, r)
        v = nums[p]
        nums[l], nums[p] = nums[p], nums[l]

        # [l+1, lt] < v, [lt+1, i) = v, [gt, r] > v
        lt, i, gt = l, l + 1, r + 1
        
        while i < gt:
            if nums[i] == v: i += 1
            elif nums[i] < v:
                lt += 1
                nums[lt], nums[i] = nums[i], nums[lt]
                i += 1
            else:
                gt -= 1
                nums[i], nums[gt] = nums[gt], nums[i]

        nums[l], nums[lt] = nums[lt], nums[l]
        self.__sort(nums, l, lt - 1)
        self.__sort(nums, gt, r)
```
